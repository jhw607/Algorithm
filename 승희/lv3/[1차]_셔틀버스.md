def to_min (s):
    h, m = s.split(':')
    return 60 * int(h) + int(m)

def to_time (i):
    q, r = divmod(i, 60)
    q = '0' + str(q) if q < 10 else str(q)
    r = '0' + str(r) if r < 10 else str(r)
    return q + ':' + r

from collections import Counter

def solution(n, t, m, timetable):
    answer = ''
    bus_time = to_min('9:00')
    last_bus_time = to_min('9:00') + (n-1) * t
    
    table = [to_min(x) for x in timetable]
    table.sort()
    
    d = {}
    
    for x in range(n):
        d[bus_time + x * t] = []
    d['None'] = []
    
    for t in table:
        flag = False
        for x in d.keys():
            if x == 'None':
                break
            if t <= x:
                d[x].append(t)
                flag = True
                break
        if not flag:
            d['None'].append(t)
    
    temp = []
    for x in d.keys():
        if temp:
            d[x] += temp
            temp = []
            
        now_m = len(d[x])
        if now_m <= m:
            continue
        
        d[x].sort()
        temp = d[x][m:]
        d[x] = d[x][:m]
      
    # print(d)
    last = list(d.keys())[-2]
    if len(d[last]) < m:
        answer = to_time(last)
    else:
        temp = Counter(d[last]).most_common()
        temp.sort()
        for t, temp_m in temp:
            
            if temp_m < m:
                m -= temp_m
            else:
                answer = to_time(t - 1)
                
    return answer
