from collections import deque

def is_ok(x1, y1, x2, y2, n, board):
    if 0 <= x1 < n and 0 <= y1 < n and 0 <= x2 < n and 0 <= y2 < n:
        if board[x1][y1] == 0 and board[x2][y2] == 0:
            return True
    return False

def is_ok_r(direction, x1, y1, x2, y2, n, board):
    result = 0
    
    if x1 == x2: # 가로
        if y1 > y2:
            y1, y2 = y2, y1
            
        if direction == 1:
            mid_x, mid_y = x1 + 1, y1
            nx, ny = mid_x, y2
            if is_ok(mid_x, mid_y, nx, ny, n, board):
                result = (nx, ny, x2, y2)
        elif direction == -1:
            mid_x, mid_y = x1 - 1, y1
            nx, ny = mid_x, y2
            if is_ok(mid_x, mid_y, nx, ny, n, board):
                result = (nx, ny, x2, y2)
    elif y1 == y2: # 세로
        if x1 > x2:
            x1, x2 = x2, x1
        if direction == 1:
            mid_x, mid_y = x1, y1 + 1
            nx, ny = x2, mid_y
            if is_ok(mid_x, mid_y, nx, ny, n, board):
                result = (nx, ny, x2, y2)
        elif direction == -1:
            mid_x, mid_y = x1, y1 - 1
            nx, ny = x2, mid_y
            if is_ok(mid_x, mid_y, nx, ny, n, board):
                result = (nx, ny, x2, y2)
    return result

def is_ok_l(direction, x1, y1, x2, y2, n, board):
    result = 0
    
    if x1 == x2: # 가로
        if y1 > y2:
            y1, y2 = y2, y1
            
        if direction == 1:
            mid_x, mid_y = x2 + 1, y2
            nx, ny = mid_x, y1
            if is_ok(mid_x, mid_y, nx, ny, n, board):
                result = (x1, y1, nx, ny)
        elif direction == -1:
            mid_x, mid_y = x2 - 1, y2
            nx, ny = mid_x, y1
            
            if is_ok(mid_x, mid_y, nx, ny, n, board):
                result = (x1, y1, nx, ny)
    elif y1 == y2: # 세로
        if x1 > x2:
            x1, x2 = x2, x1
        if direction == 1:
            mid_x, mid_y = x2, y2 + 1
            nx, ny = x1, mid_y
            if is_ok(mid_x, mid_y, nx, ny, n, board):
                result = (x1, y1, nx, ny)
        elif direction == -1:
            mid_x, mid_y = x2, y2 - 1
            nx, ny = x1, mid_y
            if is_ok(mid_x, mid_y, nx, ny, n, board):
                result = (x1, y1, nx, ny)
    return result

def solution(board):
    answer = 0
    start = [0, 0, 0, 1, 0] # a_x, a_y, b_x, b_y, t
    n = len(board)
    goal = (n-1, n-1)
    
    dx = [1, -1, 0, 0]
    dy = [0, 0, 1, -1]
    
    s = set()
    
    q = deque([start])
    
    while q:
        a_x, a_y, b_x, b_y, t = q.popleft()
        x = (a_x, a_y, b_x, b_y)
        if x in s:
            continue
        else:
            s.add(x)
        if (a_x, a_y) == goal or (b_x, b_y) == goal:
            answer = t
            break
        for i in range(4):
            na_x, na_y = a_x + dx[i], a_y + dy[i]
            nb_x, nb_y = b_x + dx[i], b_y + dy[i]
            if is_ok(na_x, na_y, nb_x, nb_y, n, board):
                q.append([na_x, na_y, nb_x, nb_y, t+1])
        for d in [1, -1]:
            temp = is_ok_r(d, a_x, a_y, b_x, b_y, n, board)
            if temp != 0:
                q.append(list(temp)+ [t+1])
        for d in [1, -1]:
            temp = is_ok_l(d, a_x, a_y, b_x, b_y, n, board)
            if temp != 0:
                q.append(list(temp)+ [t+1])

    return answer
