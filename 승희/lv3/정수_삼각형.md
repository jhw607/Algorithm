def solution(triangle):
    base = triangle[0]
    for i in range(1, len(triangle)):
        next = triangle[i]
        size = len(next)
        temp = []
        for n_i in range(size):
            if n_i == 0:
                temp.append(base[0]+next[0])
            elif n_i == size - 1:
                temp.append(base[-1]+next[-1])
            else:
                temp.append(max(base[n_i - 1] + next[n_i], base[n_i] + next[n_i]))
        base = temp
    
    return max(temp)
