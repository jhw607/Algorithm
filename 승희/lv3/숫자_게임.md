from collections import deque

def solution(A, B):
    answer = 0
    A.sort()
    B = deque(sorted(B))
    
    for a in A:
        n = len(B)
        i = 0
        while i < n:
            i += 1 
            now = B.popleft()
            if a < now:
                answer += 1
                break
            else:
                B.append(now)
            
    return answer
