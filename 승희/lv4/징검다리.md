def solution(distance, rocks, n):
    answer = 0
    rocks.append(distance)
    rocks.sort()  
    
    s = 0
    e = distance
    
    while s < e:
        mid = (s + e) // 2
        a = 0
        temp = 0
        for r in rocks:
            if r-a < mid:
                temp += 1
            else:
                a = r
        if temp <= n:
            s = mid + 1
            answer = mid
        else:
            e = mid
            
    return answer
