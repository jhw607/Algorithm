from collections import deque

def solution(places):
    answer = []
    for place in places:
        print(one(place))
    
    return answer

def one(place):
    people = []
    graph = []
    for i in range(5):
        row = place[i]
        now_list = list(row)
        graph.append(now_list)
        for j in range(5):
            if place[i][j] == 'P':
                people.append((i,j))

    dx = [-1, 1, 0, 0]
    dy = [0, 0, -1, 1]
    
    for person in people:
        x, y = person
        if graph[x][y] == 'P':
            graph[x][y] = 0
            q = deque([person])
        
        while q:
            x, y = q.popleft()
            if graph[x][y] not in range(10):
                break

            for i in range(4):
                nx = x + dx[i]
                ny = y + dy[i]
                if 0<=nx<5 and 0<=ny<5:
                    if graph[nx][ny] == 'O' or graph[nx][ny] == 'P':
                        graph[nx][ny] = graph[x][y] + 1
                        q.append((nx,ny))
                        
    max_gap = 0
    
    for i in graph:
        for x in i:
            if type(x) == int:
                max_gap = max(max_gap, x)
                        
    return max_gap
