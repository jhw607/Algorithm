from collections import deque
def solution(priorities, location):
    answer = 0
    rest = deque([(x, i) for i, x in enumerate(priorities)])
    done = deque()
    while 1:
        now = rest.popleft()
        if len([x[0] for x in rest if now[0] < x[0]]) > 0:
            rest.append(now)
        else:
            done.append(now)
            if now[1] == location:
                answer = len(done)
                break
    return answer
